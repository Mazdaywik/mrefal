--------------------------------------------------------------------------------
	Структура "Модуль -- промежуточное представление"
--------------------------------------------------------------------------------

t.Module ::=
	(t.SymTable e.ModuleContent)

e.ModuleContent ::= t.ModuleElement*

t.ModuleElement ::=
	(Function
		s.ScopeClass (e.FnName) t.QualifiedName t.FunctionBody
	)

s.ScopeClass ::=
	Entry
	Local

t.FunctionBody ::=
	( t.Sentence* )

t.Sentence ::=
	t.PatternExpr t.ResultExpr

t.PatternExpr ::=
	( t.PatternTerm* )

t.ResultExpr ::=
	( t.ResultTerm* )

t.PatternTerm ::=
	(Identifier e.Name)
	(EVar e.Index)
	(SVar e.Index)
	(TVar e.Index)
	(Literal e.Content)
	(DQWord e.Content)
	(Macro s.Value)
	(Structure e.InBrackets)
	(FnPtr t.Callable)

t.ResultTerm ::=
	t.PatternTerm
	(Function t.Callable e.Argument)
	(IndirectCall (SVar e.Index) e.Argument)

t.Callable ::=
	(Qualified t.SrcPosModule t.QualifiedName)
