t.Module ::=
	( (Module e.ModuleName)
		t.ModuleElement*
	)

t.ModuleElement ::=
	(Import e.ModuleName)
	(Function s.ScopeClass (e.FnName) t.FunctionBody)

s.ScopeClass ::=
	Entry
	Local

t.FunctionBody ::=
	t.Sentense*

t.Sentense ::=
	t.PatternExpr t.ResultExpr

t.PatternExpr ::=
	t.Term* | FunctionCall

t.ResultExpr ::=
	t.Term*

t.Term ::=
	(Identifier e.Name)
	(EVar e.Index)
	(SVar e.Index)
	(TVar e.Index)
	(Literal e.Content)
	(DQWord e.Content)
	(Macro s.Value)
	(Structure e.InBrackets)
	(Function t.Callee e.Argument)

t.Callee ::=
	(Qualified
		(Module t.LnNumModule e.ModName)
		(Function t.LnNumFunction e.FnName)
	)
	(Local
		(Function t.LnNumFunction e.FnName)
	)

t.Callee ::=
